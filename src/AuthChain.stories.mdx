## Auth Chain Generator

<Divider size="small" />

export function AuthChainStandar() {
  const [account] = useAuth()
  const [state, setState] = useState({
    method: 'GET',
    path: '/path/to/resource',
    metadata: '{}',
  })
  const [value, setValue] = useState({})
  const [signing, sign] = useAsyncTask(async () => {
    const identity = getCurrentIdentity()
    if (identity) {
      const timestamp = Date.now()
      const chain = await signPayload(
        getCurrentIdentity(),
        [
          state.method,
          state.path,
          timestamp,
          (state.metadata || '{}').toLowerCase(),
        ]
          .join(':')
          .toLowerCase()
      )
      setValue({ ...state, timestamp, chain })
    }
  }, [value])
  return (
    <Container>
      <Grid>
        {!account && <NoAuthMessage />}
        <Grid.Row>
          <Grid.Column tablet="7">
            <ArgsTable
              compact={true}
              updateArgs={(value) => {
                console.log(value)
                setState((current) => ({ ...current, ...value }))
              }}
              rows={{
                method: {
                  name: 'method',
                  control: {
                    type: 'select',
                    value: state.method,
                    options: ['GET', 'POST', 'PUT', 'PATCH', 'DELETE', 'HEAD'],
                  },
                },
                path: {
                  name: 'path',
                  control: { type: 'text', value: state.path },
                },
                metadata: {
                  name: 'metadata',
                  control: {
                    type: 'text',
                    value: state.metadata,
                    placeholder: '{}',
                  },
                },
              }}
            />
          </Grid.Column>
          <Grid.Column tablet="9">
            <Button
              primary
              size="small"
              style={{ minWidth: '200px', margin: '0 auto' }}
              disabled={!account}
              loading={signing}
              onClick={() => sign()}
            >
              Sign
            </Button>
            <Source
              language="ts"
              code={JSON.stringify(value.chain || null, null, 2)}
            />
          </Grid.Column>
        </Grid.Row>
        <AuthChainTable
          chain={value.chain}
          timestamp={value.timestamp}
          metadata={value.metadata}
        />
      </Grid>
    </Container>
  )
}

<AuthChainStandar />

## Auth Chain Raw Generator

> Generate an auth chain usign a raw payload

<Divider size="small" />

export function AuthChainRaw() {
  const [account] = useAuth()
  const [chain, setChain] = useState(null)
  const [message, setMessage] = useState('get:/path/to/resource')
  const [signing, sign] = useAsyncTask(
    async (payload) => {
      const identity = getCurrentIdentity()
      if (identity) {
        const chain = await signPayload(
          getCurrentIdentity(),
          payload.toLowerCase()
        )
        setChain(chain)
      }
    },
    [chain]
  )
  return (
    <Container>
      <Grid>
        {!account && <NoAuthMessage />}
        <Grid.Row>
          <Grid.Column tablet="7">
            <Textarea
              value={message}
              minHeight={170}
              onChange={(e, { value }) => setMessage(value || '')}
            />
          </Grid.Column>
          <Grid.Column tablet="9">
            <Button
              primary
              size="small"
              style={{ minWidth: '200px', margin: '0 auto' }}
              disabled={!account}
              loading={signing}
              onClick={() => sign(message)}
            >
              Sign
            </Button>
            <Source language="ts" code={JSON.stringify(chain, null, 2)} />
          </Grid.Column>
        </Grid.Row>
        <AuthChainTable chain={chain} />
      </Grid>
    </Container>
  )
}
